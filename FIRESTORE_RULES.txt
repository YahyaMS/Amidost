rules_version = '2';
service cloud.firestore {
  match /databases/{database}/documents {

    // Helper function to check if user is admin
    function isAdmin() {
      // Replace with your actual admin email
      return request.auth != null && request.auth.token.email == 'info@amidost.com';
    }

    // Contact form submissions
    match /contactSubmissions/{document=**} {
      // Anyone can submit contact forms
      allow create: if true;

      // Only the specific admin email can read/update/delete
      allow read, update, delete: if isAdmin();
    }

    // Blog articles
    match /articles/{articleId} {
      // Anyone can read published articles (public blog)
      allow read: if true;

      // Only the specific admin email can create, update, or delete articles
      allow create, update, delete: if isAdmin();
    }
  }
}
